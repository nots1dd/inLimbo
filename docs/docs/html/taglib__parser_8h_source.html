78// Function to parse metadata from a file
79bool TagLibParser::parseFile(const std::string& filePath, Metadata& metadata) {
80 TagLib::FileRef file(filePath.c_str());
81 std::string errMsg;
82 if (file.isNull()) {
83 errMsg = "Error: Failed to open file: " + filePath;
84 sendErrMsg(debugLogBoolStr, errMsg);
85 return false;
86 }
87
88 if (!file.tag()) {
89 errMsg = "Error: No tag information found in file: " + filePath;
90 sendErrMsg(debugLogBoolStr, errMsg);
91 return false;
92 }
93
94 TagLib::Tag* tag = file.tag();
95 metadata.title = tag->title().isEmpty() ? "Unknown Title" : tag->title().to8Bit(true);
96 metadata.artist = tag->artist().isEmpty() ? "Unknown Artist" : tag->artist().to8Bit(true);
97 metadata.album = tag->album().isEmpty() ? "Unknown Album" : tag->album().to8Bit(true);
98 metadata.genre = tag->genre().isEmpty() ? "Unknown Genre" : tag->genre().to8Bit(true);
99 metadata.comment = tag->comment().isEmpty() ? "No Comment" : tag->comment().to8Bit(true);
100 metadata.year = tag->year() == 0 ? 0 : tag->year();
101 metadata.track = tag->track() == 0 ? 0 : tag->track();
102
103 TagLib::AudioProperties *audioProperties = file.audioProperties();
104 if (audioProperties) {
105 metadata.duration = audioProperties->length(); // Duration in seconds
106 }
107
108 // Keep track of file path
109 metadata.filePath = filePath;
110
111 // Extract additional properties such as lyrics and disc number
112 TagLib::PropertyMap properties = file.file()->properties();
113 if (properties.contains("DISCNUMBER")) {
114 metadata.discNumber = properties["DISCNUMBER"].toString().toInt();
115 } else {
116 metadata.discNumber = 0;
117 }
118
119 if (properties.contains("LYRICS")) {
120 metadata.lyrics = properties["LYRICS"].toString().to8Bit(true);
121 } else {
122 metadata.lyrics = "No Lyrics";
123 }
124
125 // Populate additional properties if needed
126 for (const auto& prop : properties) {
127 std::string key = prop.first.to8Bit(true);
128 std::string value = prop.second.toString().to8Bit(true);
129 metadata.additionalProperties[key] = value;
130 }
131
132 return true;
133}

136std::unordered_map<std::string, Metadata> TagLibParser::parseFromInode(ino_t inode, const std::string& directory) {
137 std::unordered_map<std::string, Metadata> metadataMap;
138 std::string tempErrMsg;
139
140 for (const auto& entry : fs::recursive_directory_iterator(directory)) {
141 struct stat fileStat;
142 if (stat(entry.path().c_str(), &fileStat) == 0) {
143 if (fileStat.st_ino == inode) {
144 Metadata metadata;
145 if (parseFile(entry.path().string(), metadata)) {
146 metadataMap[entry.path().string()] = metadata;
147 } else {
148 tempErrMsg = "Error: Unable to parse metadata for file: " + entry.path().string();
149 sendErrMsg(debugLogBoolStr, tempErrMsg);
150 }
151 }
152 } else {
153 tempErrMsg = "Error: Unable to stat file: " + entry.path().string();
154 sendErrMsg(debugLogBoolStr, tempErrMsg);
155 }
156 }
157
158 return metadataMap;
159}
