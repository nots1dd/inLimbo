139 void parseArgs(int argc, char* argv[])
140 {
141 for (int i = 1; i < argc; ++i)
142 {
143 std::string arg = argv[i];
144
145 if (arg[0] == '-')
146 {
147 std::string flag = arg;
148 std::string value;
149
150 if (i + 1 < argc && argv[i + 1][0] != '-')
151 {
152 value = argv[++i];
153 }
154
155 // Find closest matching flag for error message
156 if (std::find(validFlags.begin(), validFlags.end(), flag) == validFlags.end())
157 {
158 hasError = true;
159
160 // Find the closest matching flag for a helpful suggestion
161 std::string closestMatch = findClosestMatch(flag);
162 if (!closestMatch.empty())
163 {
164 errorMessage = "Invalid flag: '" + flag + "'. Did you mean '" + closestMatch + "'?";
165 }
166 else
167 {
168 errorMessage = "Invalid flag: '" + flag + "'";
169 }
170 return;
171 }
172
173 args[flag] = value;
174 }
175 else
176 {
177 positionalArgs.push_back(arg);
178 }
179 }
180 }

192 std::string findClosestMatch(const std::string& invalidFlag) const
193 {
194 std::string bestMatch;
195 size_t minDist = std::string::npos;
196
197 for (const auto& validFlag : validFlags)
198 {
199 // Compute Levenshtein distance
200 size_t dist = levenshteinDistance(invalidFlag, validFlag);
201
202 // Track the flag with the smallest distance
203 if (dist < minDist)
204 {
205 minDist = dist;
206 bestMatch = validFlag;
207 }
208 }
209
210 // Only suggest if the match is reasonably close (for example, max 3 edits away)
211 return (minDist <= 3) ? bestMatch : "";
212 }
