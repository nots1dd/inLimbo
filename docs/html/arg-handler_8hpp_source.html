105 static void handleArguments(const CommandLineArgs& cmdArgs, const std::string& programName,
106 const Paths& paths)
107 {
108 const std::unordered_map<std::string, std::function<void()>> argumentHandlers = {
109 {"--help", [&]() { handleHelp(cmdArgs, programName); }},
110 {"--version", [&]() { handleVersion(); }},
111 {"--clear-cache", [&]() { handleClearCache(paths.libBinPath); }},
112 {"--show-config-file", [&]() { handleShowConfig(paths.configPath); }},

114 {"--show-dbus-name", [&]() { handleShowDBusName(); }},
115 {"--update-cache-run", [&]() { handleUpdateCacheRun(paths.libBinPath); }}};
116
117 for (const auto& [flag, handler] : argumentHandlers)
118 {
119 if (cmdArgs.hasFlag(flag))
120 {
121 handler();
122 if (!shouldRunApp)
123 exit(0);
124 }
125 }
126 }
