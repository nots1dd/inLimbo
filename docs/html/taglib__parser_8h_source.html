138// Function to parse metadata from a file
139bool TagLibParser::parseFile(const std::string& filePath, Metadata& metadata) {
140 TagLib::FileRef file(filePath.c_str());
141 std::string errMsg;
142 if (file.isNull()) {
143 errMsg = "Error: Failed to open file: " + filePath;
144 sendErrMsg(debugLogBoolStr, errMsg);
145 return false;
146 }
147
148 if (!file.tag()) {
149 errMsg = "Error: No tag information found in file: " + filePath;
150 sendErrMsg(debugLogBoolStr, errMsg);
151 return false;
152 }
153
154 TagLib::Tag* tag = file.tag();
155 metadata.title = tag->title().isEmpty() ? "Unknown Title" : tag->title().to8Bit(true);
156 metadata.artist = tag->artist().isEmpty() ? "Unknown Artist" : tag->artist().to8Bit(true);
157 metadata.album = tag->album().isEmpty() ? "Unknown Album" : tag->album().to8Bit(true);
158 metadata.genre = tag->genre().isEmpty() ? "Unknown Genre" : tag->genre().to8Bit(true);
159 metadata.comment = tag->comment().isEmpty() ? "No Comment" : tag->comment().to8Bit(true);
160 metadata.year = tag->year() == 0 ? 0 : tag->year();
161 metadata.track = tag->track() == 0 ? 0 : tag->track();
162
163 TagLib::AudioProperties *audioProperties = file.audioProperties();
164 if (audioProperties) {
165 metadata.duration = audioProperties->length(); // Duration in seconds
166 }
167
168 // Keep track of file path
169 metadata.filePath = filePath;
170
171 // Extract additional properties such as lyrics and disc number
172 TagLib::PropertyMap properties = file.file()->properties();
173 if (properties.contains("DISCNUMBER")) {
174 metadata.discNumber = properties["DISCNUMBER"].toString().toInt();
175 } else {
176 metadata.discNumber = 0;
177 }
178
179 if (properties.contains("LYRICS")) {
180 metadata.lyrics = properties["LYRICS"].toString().to8Bit(true);
181 } else {
182 metadata.lyrics = "No Lyrics";
183 }
184
185 // Populate additional properties if needed
186 for (const auto& prop : properties) {
187 std::string key = prop.first.to8Bit(true);
188 std::string value = prop.second.toString().to8Bit(true);
189 metadata.additionalProperties[key] = value;
190 }
191
192 return true;
193}

195// Function to parse metadata based on inode
196std::unordered_map<std::string, Metadata> TagLibParser::parseFromInode(ino_t inode, const std::string& directory) {
197 std::unordered_map<std::string, Metadata> metadataMap;
198 std::string tempErrMsg;
199
200 for (const auto& entry : fs::recursive_directory_iterator(directory)) {
201 struct stat fileStat;
202 if (stat(entry.path().c_str(), &fileStat) == 0) {
203 if (fileStat.st_ino == inode) {
204 Metadata metadata;
205 if (parseFile(entry.path().string(), metadata)) {
206 metadataMap[entry.path().string()] = metadata;
207 } else {
208 tempErrMsg = "Error: Unable to parse metadata for file: " + entry.path().string();
209 sendErrMsg(debugLogBoolStr, tempErrMsg);
210 }
211 }
212 } else {
213 tempErrMsg = "Error: Unable to stat file: " + entry.path().string();
214 sendErrMsg(debugLogBoolStr, tempErrMsg);
215 }
216 }
217
218 return metadataMap;
219}

267bool extractThumbnail(const std::string& audioFilePath, const std::string& outputImagePath) {
268 // Open the file using TagLib
269 TagLib::MPEG::File file(audioFilePath.c_str());
270 if (!file.isValid()) {
271 std::cerr << "Error: Could not open audio file." << std::endl;
272 return false;
273 }
274
275 // Ensure the file has ID3v2 tags
276 TagLib::ID3v2::Tag* id3v2Tag = file.ID3v2Tag();
277 if (!id3v2Tag) {
278 std::cerr << "Error: No ID3v2 tags found in the audio file." << std::endl;
279 return false;
280 }
281
282 // Search for the APIC (Attached Picture) frame
283 const TagLib::ID3v2::FrameList& frameList = id3v2Tag->frameListMap()["APIC"];
284 if (frameList.isEmpty()) {
285 std::cerr << "Error: No embedded album art found in the audio file." << std::endl;
286 return false;
287 }
288
289 // Extract the first APIC frame (album art)
290 auto* apicFrame = dynamic_cast<TagLib::ID3v2::AttachedPictureFrame*>(frameList.front());
291 if (!apicFrame) {
292 std::cerr << "Error: Failed to retrieve album art." << std::endl;
293 return false;
294 }
295
296 // Get the picture data and MIME type
297 const auto& pictureData = apicFrame->picture();
298 const std::string mimeType = apicFrame->mimeType().toCString(true);
299
300 // Save the picture data to a file
301 std::ofstream outputFile(outputImagePath, std::ios::binary);
302 if (!outputFile) {
303 std::cerr << "Error: Could not create output image file." << std::endl;
304 return false;
305 }
306
307 outputFile.write(reinterpret_cast<const char*>(pictureData.data()), pictureData.size());
308 outputFile.close();
309
310 return true;
311}
