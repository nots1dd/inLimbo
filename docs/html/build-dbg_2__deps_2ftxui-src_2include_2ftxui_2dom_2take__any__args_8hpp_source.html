7// IWYU pragma: private, include "ftxui/dom/elements.hpp"
8#include <type_traits>
9
10namespace ftxui {
11
12template <class T>
13void Merge(Elements& /*container*/, T /*element*/) {}
14
15template <>
16inline void Merge(Elements& container, Element element) {
17 container.push_back(std::move(element));
18}
19
20template <>
21inline void Merge(Elements& container, Elements elements) {
22 for (auto& element : elements)
23 container.push_back(std::move(element));
24}
25
26// Turn a set of arguments into a vector.
27template <class... Args>
28Elements unpack(Args... args) {
29 std::vector<Element> vec;
30 (Merge(vec, std::move(args)), ...);
31 return vec;
32}
33
34// Make |container| able to take any number of argments.
35#define TAKE_ANY_ARGS(container) \
36 template <class... Args> \
37 Element container(Args... children) { \
38 return container(unpack(std::forward<Args>(children)...)); \
39 }
40
41TAKE_ANY_ARGS(vbox)
42TAKE_ANY_ARGS(hbox)
43TAKE_ANY_ARGS(dbox)
44TAKE_ANY_ARGS(hflow)
45} // namespace ftxui
