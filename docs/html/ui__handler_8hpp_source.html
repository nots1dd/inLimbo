413 {
414 current_inodes.clear();

416 album_name_indices.clear();
417
418 if (library.count(artist) > 0)

420 // Group songs by album and year
421 std::map<std::string, std::map<unsigned int, std::map<unsigned int, Song>>> albums;
422 for (const auto& album_pair : library.at(artist))
423 {
424 const std::string& album_name = album_pair.first;
425 for (const auto& disc_pair : album_pair.second)
426 {
427 for (const auto& track_pair : disc_pair.second)
428 {
429 albums[album_name][disc_pair.first][track_pair.first] = track_pair.second;
430 }
431 }
432 }
433
434 // Format the album and song display
435 for (const auto& [album_name, discs] : albums)
436 {
437 // Get year from the first song in the album
438 const Song& first_song = discs.begin()->second.begin()->second;
439 current_song_elements.push_back(
440 renderAlbumName(album_name, first_song.metadata.year, global_colors.album_name_bg));
441 album_name_indices.insert(current_song_elements.size() - 1);
442
443 for (const auto& [disc_number, tracks] : discs)
444 {
445 for (const auto& [track_number, song] : tracks)
446 {
447 std::string disc_track_info =

449 current_inodes.push_back(song.inode);
