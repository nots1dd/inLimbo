211ftxui::Color parseHexColor(const std::string& hex)
212{
213 // Validate hex format
214 std::regex hex_regex("^#([0-9a-fA-F]{6})$"); // Matches #RRGGBB
215 std::smatch match;
216
217 if (!std::regex_match(hex, match, hex_regex))
218 {
219 std::cerr << "Error: Invalid hexadecimal color format: '" << hex
220 << "'. Use a string like `#RRGGBB`." << std::endl;
221 std::exit(EXIT_FAILURE);
222 }
223
224 // Extract RGB components from the validated hex string
225 int r = std::stoi(hex.substr(1, 2), nullptr, 16);
226 int g = std::stoi(hex.substr(3, 2), nullptr, 16);
227 int b = std::stoi(hex.substr(5, 2), nullptr, 16);
228
229 return ftxui::Color::RGB(r, g, b);
230}
